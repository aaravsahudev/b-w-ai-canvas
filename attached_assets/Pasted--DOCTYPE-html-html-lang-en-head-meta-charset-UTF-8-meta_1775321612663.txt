// -- SAFE JSON PARSER --
function safeParseJSON(raw) {
  try { return JSON.parse(raw); } catch(e) {}
  const start = raw.indexOf('{');
  if (start === -1) return null;
  let depth = 0, end = -1;
  for (let i = start; i < raw.length; i++) {
    if (raw[i] === '{') depth++;
    else if (raw[i] === '}') { depth--; if (depth === 0) { end = i; break; } }
  }
  const chunk = end !== -1 ? raw.slice(start, end + 1) : raw.slice(start);
  try { return JSON.parse(chunk); } catch(e) {}
  let fixed = chunk;
  let opens = 0, squareOpens = 0;
  let inStr = false, escape = false;
  for (const ch of fixed) {
    if (escape) { escape = false; continue; }
    if (ch === '\\') { escape = true; continue; }
    if (ch === '"') { inStr = !inStr; continue; }
    if (inStr) continue;
    if (ch === '{') opens++;
    else if (ch === '}') opens--;
    else if (ch === '[') squareOpens++;
    else if (ch === ']') squareOpens--;
  }
  fixed = fixed.replace(/,\s*$/, '');
  for (let i = 0; i < squareOpens; i++) fixed += ']';
  for (let i = 0; i < opens; i++) fixed += '}';
  try { return JSON.parse(fixed); } catch(e) {}
  return null;
}

function drawRivalChart(rivals) {
  const canvas = document.getElementById('rivalChart');
  if (!canvas) return;
  const dpr = window.devicePixelRatio || 1;
  const W = canvas.parentElement.clientWidth;
  const H = 200;
  canvas.width = W * dpr;
  canvas.height = H * dpr;
  canvas.style.width = W + 'px';
  canvas.style.height = H + 'px';
  const ctx = canvas.getContext('2d');
  ctx.scale(dpr, dpr);

  const padL=52, padR=20, padT=24, padB=48;
  const cW=W-padL-padR, cH=H-padT-padB;
  const n=rivals.length, groupW=cW/n, barW=Math.min(groupW*0.32,48), gap=8;
  const maxGrowth=Math.max(...rivals.map(r=>r.yoy_growth))*1.2, maxThreat=100;

  let progress=0;
  const start=performance.now();

  function frame(now){
    progress=Math.min((now-start)/800,1);
    const ease=1-Math.pow(1-progress,3);
    ctx.clearRect(0,0,W,H);

    ctx.font=`300 9px 'DM Mono', monospace`;
    for(let i=0;i<=4;i++){
      const y=padT+cH-(i/4)*cH;
      ctx.strokeStyle='rgba(255,255,255,0.05)'; ctx.lineWidth=1;
      ctx.beginPath(); ctx.moveTo(padL,y); ctx.lineTo(padL+cW,y); ctx.stroke();
      ctx.fillStyle='rgba(74,74,85,1)'; ctx.textAlign='right';
      ctx.fillText(Math.round(maxGrowth*i/4)+'%',padL-6,y+3);
    }

    rivals.forEach((r,i)=>{
      const cx=padL+i*groupW+groupW/2;
      const gH=(r.yoy_growth/maxGrowth)*cH*ease, gx=cx-gap/2-barW, gy=padT+cH-gH;
      const gradG=ctx.createLinearGradient(gx,gy,gx,padT+cH);
      gradG.addColorStop(0,'rgba(247,245,240,0.9)'); gradG.addColorStop(1,'rgba(247,245,240,0.06)');
      ctx.fillStyle=gradG; ctx.fillRect(gx,gy,barW,gH);

      const tH=(r.threat_level/maxThreat)*cH*ease, tx=cx+gap/2, ty=padT+cH-tH;
      const tc=threatColor(r.threat_level);
      const gradT=ctx.createLinearGradient(tx,ty,tx,padT+cH);
      gradT.addColorStop(0,tc); gradT.addColorStop(1,tc.replace(')',',.08)').replace('rgb','rgba'));
      ctx.fillStyle=gradT; ctx.fillRect(tx,ty,barW,tH);

      if(progress>0.7){
        const a=Math.min((progress-0.7)/0.3,1); ctx.globalAlpha=a;
        ctx.font=`500 9px 'DM Mono', monospace`; ctx.textAlign='center';
        ctx.fillStyle='rgba(247,245,240,0.7)'; ctx.fillText('+'+r.yoy_growth+'%',gx+barW/2,gy-5);
        ctx.fillStyle=tc; ctx.fillText(r.threat_level,tx+barW/2,ty-5);
        ctx.font=`700 11px 'Syne', sans-serif`; ctx.fillStyle='rgba(247,245,240,0.7)';
        ctx.fillText(r.name,cx,padT+cH+18);
        ctx.font=`300 9px 'DM Mono', monospace`; ctx.fillStyle='rgba(74,74,85,1)';
        ctx.fillText((new Date().getFullYear()-r.founded)+'yr',cx,padT+cH+32);
        ctx.globalAlpha=1;
      }
    });

    ctx.strokeStyle='rgba(255,255,255,0.1)'; ctx.lineWidth=1;
    ctx.beginPath(); ctx.moveTo(padL,padT+cH); ctx.lineTo(padL+cW,padT+cH); ctx.stroke();

    if(progress>0.8){
      const la=Math.min((progress-0.8)/0.2,1); ctx.globalAlpha=la;
      ctx.font=`300 9px 'DM Mono', monospace`; ctx.textAlign='left';
      ctx.fillStyle='rgba(247,245,240,0.55)'; ctx.fillRect(padL+cW-120,padT+2,9,9);
      ctx.fillStyle='rgba(74,74,85,1)'; ctx.fillText('YoY Growth',padL+cW-107,padT+10);
      ctx.fillStyle=threatColor(75); ctx.fillRect(padL+cW-120,padT+18,9,9);
      ctx.fillStyle='rgba(74,74,85,1)'; ctx.fillText('Threat Level',padL+cW-107,padT+26);
      ctx.globalAlpha=1;
    }
    if(progress<1) requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}

function drawRiskGauge(rp) {
  const canvas=document.getElementById('riskGauge');
  if(!canvas) return;
  const score=rp.overall||50, color=riskColor(score);
  const W=200, H=116, dpr=window.devicePixelRatio||1;
  canvas.width=W*dpr; canvas.height=H*dpr;
  canvas.style.width=W+'px'; canvas.style.height=H+'px';
  const ctx=canvas.getContext('2d');
  ctx.scale(dpr,dpr);
  const cx=W/2, cy=H-10, r=88, rInner=62;
  const startA=Math.PI, endA=2*Math.PI;
  const zones=[
    {from:0,to:20,color:'rgba(168,197,160,0.18)'},{from:20,to:40,color:'rgba(200,200,154,0.18)'},
    {from:40,to:60,color:'rgba(200,169,110,0.18)'},{from:60,to:80,color:'rgba(200,130,90,0.18)'},
    {from:80,to:100,color:'rgba(200,96,96,0.18)'}

  const dur=1000, st=performance.now();
  function frame(now){
    prog=Math.min((now-st)/dur,1);
    const ease=1-Math.pow(1-prog,3);
    const fillAngle=startA+(score/100)*Math.PI*ease;
    ctx.clearRect(0,0,W,H);

    ctx.beginPath(); ctx.arc(cx,cy,r,startA,endA); ctx.arc(cx,cy,rInner,endA,startA,true);
    ctx.closePath(); ctx.fillStyle='rgba(37,37,41,0.9)'; ctx.fill();

    zones.forEach(z=>{
      const a1=startA+(z.from/100)*Math.PI, a2=startA+(z.to/100)*Math.PI;
      ctx.beginPath(); ctx.arc(cx,cy,r-1,a1,a2); ctx.arc(cx,cy,rInner+1,a2,a1,true);
      ctx.closePath(); ctx.fillStyle=z.color; ctx.fill();
    });

    if(score>0){
      ctx.beginPath(); ctx.arc(cx,cy,r-2,startA,fillAngle); ctx.arc(cx,cy,rInner+2,fillAngle,startA,true);
      ctx.closePath(); ctx.fillStyle=color; ctx.globalAlpha=0.85; ctx.fill(); ctx.globalAlpha=1;
    }

    const needleA=startA+(score/100)*Math.PI*ease;
    const nx=cx+Math.cos(needleA)*(rInner+(r-rInner)/2), ny=cy+Math.sin(needleA)*(rInner+(r-rInner)/2);
    ctx.beginPath(); ctx.arc(nx,ny,5,0,Math.PI*2); ctx.fillStyle=color; ctx.fill();
    ctx.beginPath(); ctx.arc(nx,ny,3,0,Math.PI*2); ctx.fillStyle='#fff'; ctx.fill();

    ctx.font=`300 8px 'DM Mono', monospace`; ctx.textAlign='center';
    ctx.fillStyle='rgba(74,74,85,0.8)';
    [[' 0',0],['25',25],['50',50],['75',75],['100',100]].forEach(([lbl,pct])=>{
      const a=startA+(pct/100)*Math.PI;
      ctx.fillText(lbl, cx+Math.cos(a)*(r+10), cy+Math.sin(a)*(r+10)+3);
    });
    if(prog<1) requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}

function drawChart(data) {
  const canvas=document.getElementById('growthChart');
  if(!canvas) return;
  const dpr=window.devicePixelRatio||1;
  const W=canvas.parentElement.clientWidth-52, H=220;
  canvas.width=W*dpr; canvas.height=H*dpr;
  canvas.style.width=W+'px'; canvas.style.height=H+'px';
  const ctx=canvas.getContext('2d');
  ctx.scale(dpr,dpr);

  const padL=42, padR=16, padT=28, padB=38;
  const chartW=W-padL-padR, chartH=H-padT-padB;
  const n=data.length, groupW=chartW/n, barW=Math.min(groupW*0.28,32), gap=6;
  const maxRev=Math.max(...data.map(d=>d.revenue))*1.15;
  const maxUsr=Math.max(...data.map(d=>d.users))*1.15;
  const steps=4;

  let progress=0;
  const dur=900, start=performance.now();

  function frame(now){
    progress=Math.min((now-start)/dur,1);
    const ease=1-Math.pow(1-progress,3);
    ctx.clearRect(0,0,W,H);

    ctx.font=`300 9px 'DM Mono', monospace`; ctx.textAlign='right';
    for(let i=0;i<=steps;i++){
      const y=padT+chartH-(i/steps)*chartH, val=Math.round(maxRev*i/steps);
      ctx.strokeStyle='rgba(255,255,255,0.06)'; ctx.lineWidth=1;
      ctx.beginPath(); ctx.moveTo(padL,y); ctx.lineTo(padL+chartW,y); ctx.stroke();
      ctx.fillStyle='rgba(74,74,85,1)';
      ctx.fillText(val>=1000?(val/1000).toFixed(1)+'M':val+'K',padL-6,y+3);
    }

    data.forEach((d,i)=>{
      const cx=padL+i*groupW+groupW/2;
      const revH=(d.revenue/maxRev)*chartH*ease, rx=cx-gap/2-barW, ry=padT+chartH-revH;
      const gRev=ctx.createLinearGradient(rx,ry,rx,padT+chartH);
      gRev.addColorStop(0,'rgba(247,245,240,1)'); gRev.addColorStop(1,'rgba(247,245,240,0.07)');
      ctx.fillStyle=gRev; ctx.fillRect(rx,ry,barW,revH);

      const usrH=(d.users/maxUsr)*chartH*ease, ux=cx+gap/2, uy=padT+chartH-usrH;
      const gUsr=ctx.createLinearGradient(ux,uy,ux,padT+chartH);
      gUsr.addColorStop(0,'rgba(200,169,110,0.55)'); gUsr.addColorStop(1,'rgba(200,169,110,0.06)');
      ctx.fillStyle=gUsr; ctx.fillRect(ux,uy,barW,usrH);

      if(progress>0.85&&i>0&&d.growth_pct>0){
        ctx.globalAlpha=Math.min((progress-0.85)/0.15,1);
        ctx.font=`500 9px 'DM Mono', monospace`; ctx.textAlign='center';
        ctx.fillStyle='rgba(200,169,110,0.9)'; ctx.fillText('+'+d.growth_pct+'%',cx,ry-6);
        ctx.globalAlpha=1;
      }
      if(progress>0.4){
        ctx.globalAlpha=Math.min((progress-0.4)/0.3,1);
        ctx.font=`300 9px 'DM Mono', monospace`; ctx.textAlign='center';
        ctx.fillStyle='rgba(74,74,85,1)'; ctx.fillText(d.year,cx,padT+chartH+20);
        ctx.font=`400 9px 'DM Mono', monospace`; ctx.fillStyle='rgba(247,245,240,0.45)';
        const rv=d.revenue>=1000?(d.revenue/1000).toFixed(1)+'M':d.revenue+'K';
        ctx.fillText('$'+rv,rx+barW/2,ry-(i>0&&d.growth_pct>0?22:6));
        ctx.globalAlpha=1;
      }
    });

    ctx.strokeStyle='rgba(255,255,255,0.12)'; ctx.lineWidth=1;
    ctx.beginPath(); ctx.moveTo(padL,padT+chartH); ctx.lineTo(padL+chartW,padT+chartH); ctx.stroke();
    if(progress<1) requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}
