<script>
  const msgsEl    = document.getElementById('msgs');
  const inputEl   = document.getElementById('userInput');
  const sendBtn   = document.getElementById('sendBtn');
  const histListEl = document.getElementById('histList');

  const BOT_SVG = `<svg width="13" height="13" fill="none" stroke="#c8c8c8" stroke-width="1.5" viewBox="0 0 24 24"><rect x="2" y="7" width="20" height="14" rx="1"/><path d="M16 7V5a2 2 0 0 0-2-2h-4a2 2 0 0 0-2 2v2"/><line x1="12" y1="12" x2="12" y2="16"/><line x1="10" y1="14" x2="14" y2="14"/></svg>`;
  const USR_SVG = `<svg width="13" height="13" fill="none" stroke="#0a0a0a" stroke-width="1.5" viewBox="0 0 24 24"><circle cx="12" cy="8" r="4"/><path d="M4 20c0-4 3.6-7 8-7s8 3 8 7"/></svg>`;

  let sessions  = [];   // [{id, title, time, apiHistory, domSnap}]
  let activeId  = null;
  let busy      = false;

  /* -- Utilities -- */
  function nowTime() {
    const d = new Date();
    return d.getHours().toString().padStart(2,'0') + ':' + d.getMinutes().toString().padStart(2,'0');
  }

  function activeSession() { return sessions.find(s => s.id === activeId); }

  function snapDOM() {
    const s = activeSession();
    if (s) s.domSnap = msgsEl.innerHTML;
  }

  /* -- Sidebar render -- */
  function renderSidebar() {
    histListEl.innerHTML = '';
    [...sessions].reverse().forEach(s => {
      const div = document.createElement('div');
      div.className = 'hist-item' + (s.id === activeId ? ' active' : '');
      div.innerHTML = `<span class="hi-title">${s.title}</span><span class="hi-time">${s.time}</span>`;
      div.onclick = () => switchSession(s.id);
      histListEl.appendChild(div);
    });
  }

  /* -- Session switching -- */
  function switchSession(id) {
    snapDOM();
    activeId = id;
    const s = activeSession();
    msgsEl.innerHTML = s.domSnap || welcomeHTML();
    msgsEl.scrollTop = msgsEl.scrollHeight;
    renderSidebar();
  }

  /* -- New chat -- */
  function newChat() {
    snapDOM();
    const id = Date.now();
    sessions.push({ id, title: 'New Chat', time: nowTime(), apiHistory: [], domSnap: welcomeHTML() });
    activeId = id;
    msgsEl.innerHTML = welcomeHTML();
    renderSidebar();
    inputEl.focus();
  }

  function welcomeHTML() {
    return `<div class="row ai"><div class="av">${BOT_SVG}</div><div class="bub">Hi there! I'm QWS Agent V1 — tell me your problem and I'll help you solve it.</div></div>`;
  }

  /* -- Message helpers -- */
  function appendMsg(role, text) {
    const row = document.createElement('div');
    row.className = 'row ' + role;
    row.innerHTML = `<div class="av">${role === 'ai' ? BOT_SVG : USR_SVG}</div><div class="bub">${text.replace(/\n/g,'<br>')}</div>`;
    msgsEl.appendChild(row);
    msgsEl.scrollTop = msgsEl.scrollHeight;
  }

  function showTyping() {
    const row = document.createElement('div');
    row.className = 'row ai'; row.id = 'typing';
    row.innerHTML = `<div class="av">${BOT_SVG}</div><div class="bub"><div class="dots"><span></span><span></span><span></span></div></div>`;
    msgsEl.appendChild(row);
    msgsEl.scrollTop = msgsEl.scrollHeight;
  }

  function removeTyping() { document.getElementById('typing')?.remove(); }

  /* -- Send -- */
  inputEl.addEventListener('input', () => {
    inputEl.style.height = 'auto';
    inputEl.style.height = Math.min(inputEl.scrollHeight, 100) + 'px';
  });

  inputEl.addEventListener('keydown', e => {
    if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMsg(); }
  });

  async function sendMsg() {
    const text = inputEl.value.trim();
    if (!text || busy) return;
    busy = true;
    sendBtn.disabled = true;
    inputEl.value = '';
    inputEl.style.height = 'auto';

    const sess = activeSession();

    // Auto-title from first user message
    if (sess.apiHistory.length === 0) {
      sess.title = text.length > 26 ? text.slice(0, 26) + '…' : text;
    }

    appendMsg('user', text);
    sess.apiHistory.push({ role: 'user', content: text });
    showTyping();
    snapDOM();
    renderSidebar();

    try {
      const res = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1000,
          system: 'You are QWS Agent V1, the AI assistant for Quick Web Stack. Help users with their web development problems clearly and concisely.',
          messages: sess.apiHistory
        })
      });
      const data = await res.json();
      removeTyping();
      const reply = data.content?.[0]?.text || "Sorry, I couldn't respond.";
      sess.apiHistory.push({ role: 'assistant', content: reply });
      appendMsg('ai', reply);
    } catch {
      removeTyping();
      appendMsg('ai', 'Something went wrong. Please try again.');
    }

    snapDOM();
    busy = false;
    sendBtn.disabled = false;
    inputEl.focus();
  }

  /* -- Init -- */
  (function init() {
    const id = Date.now();
    sessions.push({ id, title: 'New Chat', time: nowTime(), apiHistory: [], domSnap: welcomeHTML() });
    activeId = id;
    msgsEl.innerHTML = welcomeHTML();
    renderSidebar();
  })();
</script>
