async function callAI(userPrompt, systemOverride){
  const sys=systemOverride||`You are an elite presentation strategist. Return ONLY valid JSON array. No markdown, no preamble.`;
  const res=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:5000,system:sys,messages:[{role:'user',content:userPrompt}]})});
  if(!res.ok){const j=await res.json().catch(()=>({}));throw new Error(j.error?.message||'API '+res.status)}
  const d=await res.json();
  const raw=d.content.map(b=>b.text||'').join('');
  const clean=raw.replace(/```json\s*/gi,'').replace(/```\s*/g,'').trim();
  let arr;
  try{arr=JSON.parse(clean)}catch{const m=clean.match(/\[[\s\S]*\]/);if(m)arr=JSON.parse(m[0]);else throw new Error('JSON parse failed')}
  if(!Array.isArray(arr))throw new Error('Expected JSON array');
  return arr;
}

function saveEdits(){
  const sd=S.slides[S.cur];
  sd.title=document.getElementById('ep-title')?.value||sd.title;
  const sub=document.getElementById('ep-subtitle');if(sub)sd.subtitle=sub.value;
  const hl=document.getElementById('ep-highlight');if(hl)sd.highlight=hl.value;
  const sn=document.getElementById('ep-notes');if(sn)sd.speaker_note=sn.value;

  // body bullets
  const bodyList=document.getElementById('ep-body-list');
  if(bodyList){
    const rows=bodyList.querySelectorAll('textarea[id^="ep-body-"]');
    sd.body=Array.from(rows).map(r=>r.value).filter(v=>v.trim());
  }
  // stats
  if(sd.type==='stats'){
    sd.stats=sd.stats||[];
    for(let i=0;i<3;i++){
      if(!sd.stats[i])sd.stats[i]={};
      const v=document.getElementById(`ep-stat-val-${i}`);const l=document.getElementById(`ep-stat-lbl-${i}`);
      if(v)sd.stats[i].value=v.value;if(l)sd.stats[i].label=l.value;
    }
  }
  // chart
  if(sd.type==='chart'){
    if(!sd.chart)sd.chart={};
    const ct=document.getElementById('ep-chart-title');if(ct)sd.chart.title=ct.value;
    const cv=document.getElementById('ep-chart-vals');if(cv)sd.chart.values=cv.value.split(',').map(v=>parseInt(v.trim())||0);
    const cl=document.getElementById('ep-chart-labs');if(cl)sd.chart.labels=cl.value.split(',').map(v=>v.trim());
  }
  // performance
  if(sd.type==='performance'){
    sd.performance=sd.performance||[];
    for(let i=0;i<4;i++){
      if(!sd.performance[i])sd.performance[i]={};
      const l=document.getElementById(`ep-kpi-lbl-${i}`);const v=document.getElementById(`ep-kpi-val-${i}`);
      if(l)sd.performance[i].label=l.value;if(v)sd.performance[i].value=parseInt(v.value)||0;
    }
  }

  S.editedSlides.add(S.cur);
  renderView();buildStrip();
  showToast('✓ Slide saved!');
}

Rewrite this slide according to the instruction. Keep the same "type" unless the instruction asks to change it.

function renderSlide(sd,t){
  const type=sd.type||'content';
  const b=Array.isArray(sd.body)?sd.body:(typeof sd.body==='string'?[sd.body]:[]);
  switch(type){
    case'title':      return sTITLE(sd,t);
    case'intro':      return sINTRO(sd,t,b);
    case'chart':      return sCHART(sd,t,b);
    case'performance':return sPERF(sd,t,b);
    case'stats':      return sSTATS(sd,t);
    case'quote':      return sQUOTE(sd,t,b);
    case'conclusion': return sCONC(sd,t,b);
    case'comparison': return sCOMP(sd,t,b);
    default:          return sCONTENT(sd,t,b);
